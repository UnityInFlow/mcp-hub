**1. Implement the SSE log stream handler in src/web/sse.rs:**

```rust
use axum::{
    extract::{Query, State},
    response::sse::{Event, KeepAlive, Sse},
};
use futures_util::Stream;
use serde::Deserialize;
use std::{convert::Infallible, sync::Arc};
use tokio_stream::{wrappers::BroadcastStream, StreamExt};

/// Query parameters for the SSE log stream endpoint.
#[derive(Deserialize)]
pub struct LogParams {
    /// Filter logs to this server name. If absent, stream all servers.
    pub server: Option<String>,
}

/// SSE endpoint that streams live log lines.
///
/// Bridges `LogAggregator.subscribe()` (broadcast::Receiver) to an SSE
/// response via `BroadcastStream`. Lagged events are silently dropped
/// (Pitfall 2) -- the HTMX SSE extension reconnects automatically.
///
/// Per D-10: Uses SSE event name "log" for `sse-swap="log"` in template.
/// Per D-11: Filters by server name when `?server=name` query param present.
pub async fn log_stream_handler(
    State(state): State<Arc<super::WebState>>,
    Query(params): Query<LogParams>,
) -> Sse<impl Stream<Item = Result<Event, Infallible>>> {
    let rx = state.log_agg.subscribe();
    let filter_server = params.server;

    let stream = BroadcastStream::new(rx)
        .filter_map(move |result| {
            let filter = filter_server.clone();
            async move {
                let line = result.ok()?; // Drop lagged errors silently (Pitfall 2)
                if let Some(ref name) = filter {
                    if &line.server != name {
                        return None;
                    }
                }
                let ts = crate::logs::format_system_time(line.timestamp);
                let data = format!(
                    "<div class=\"log-line\"><span style=\"color: #569cd6;\">[{}]</span> <span style=\"color: #808080;\">{}</span> {}</div>",
                    line.server, ts, line.message
                );
                Some(Ok(Event::default()
                    .event("log")
                    .data(data)))
            }
        });

    Sse::new(stream).keep_alive(
        KeepAlive::new()
            .interval(std::time::Duration::from_secs(15))
    )
}
```

Key implementation details:
- Event name is `"log"` matching `sse-swap="log"` in the template
- Data is an HTML fragment (a div with class `log-line`) that HTMX swaps directly into the container
- Server name gets a colored span, timestamp gets a muted span
- `KeepAlive` interval at 15 seconds prevents connection timeout
- `filter_map` drops `BroadcastStreamRecvError::Lagged` silently via `.ok()?`

The `logs_page` handler supports `?server=name` query param to pre-select a filter:
```rust
pub async fn logs_page(
    State(state): State<Arc<super::WebState>>,
    Query(params): Query<super::sse::LogParams>,
) -> LogsPage {
    let handles = state.handles.lock().await;
    let servers: Vec<ServerFilterPill> = handles.iter().map(|h| {
        ServerFilterPill {
            name: h.name.clone(),
            active: params.server.as_ref().map_or(false, |s| s == &h.name),
        }
    }).collect();
    let sse_url = match &params.server {
        Some(name) => format!("/logs/stream?server={name}"),
        None => "/logs/stream".to_string(),
    };
    LogsPage {
        active_tab: "logs".into(),
        servers,
        active_filter: params.server.clone(),
        sse_url,
    }
}
```
